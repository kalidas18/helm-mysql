apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: {{ .Release.Name }}-mysql
spec:
  selector:
    matchLabels:
      app: mysql-db 
  serviceName: headless-mysql
  replicas: {{ .Values.replicaCount }}
  template:
    metadata:
      labels:
        app: mysql-db
      annotations:
	    vault.hashicorp.com/agent-inject: "true"
	    vault.hashicorp.com/agent-inject-secret-app: "secret/mysql"
	    vault.hashicorp.com/role: "app"
    spec:
      serviceAccountName: vault
      containers:
      - name: mysql
        image: {{ .Values.image }}
        ports:
        - containerPort: 3306
          name: mysql
        volumeMounts:
        - name: mysql-volume
          mountPath: /var/lib/mysql
  volumeClaimTemplates:
  - metadata:
      name: mysql-volume
    spec:
      accessModes: 
	- ReadWriteOnce
      resources:
        requests:
          storage: 1Gi
